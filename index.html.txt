    async function obtenerReplays() {
        const { data, error } = await supabase.storage.from('replays').list('', {
            limit: 20,
            sortBy: { column: 'created_at', order: 'desc' }
        });

        if (error) {
            console.error("Error:", error);
            return;
        }

        const container = document.getElementById('replays-list');
        container.innerHTML = "";

        data.forEach(file => {
            // Obtenemos la URL pública del archivo
            const { data: urlData } = supabase.storage.from('replays').getPublicUrl(file.name);
            const publicUrl = urlData.publicUrl;

            container.innerHTML += `
                <div class="card" onclick="ver('${publicUrl}')">
                    <h3>${file.name.replace('.hbr2', '')}</h3>
                    <p>Fecha: ${new Date(file.created_at).toLocaleDateString()}</p>
                    <button style="width:100%; background:var(--accent); border:none; padding:8px; border-radius:5px; margin-top:10px; font-weight:bold; cursor:pointer;">VER AHORA</button>
                </div>
            `;
        });
    }

    function ver(url) {
        document.getElementById('modal').style.display = 'flex';
        // El visor de Haxball necesita la URL codificada
        const visor = `https://www.haxball.com/replay?url=${encodeURIComponent(url)}`;
        document.getElementById('video-box').innerHTML = `<iframe src="${visor}"></iframe>`;
    }

    function cerrar() {
        document.getElementById('modal').style.display = 'none';
        document.getElementById('video-box').innerHTML = "";
    }

    obtenerReplays();
