<script>
    // --- TUS DATOS DE SUPABASE YA CARGADOS ---
    const SUPABASE_URL = "https://vilysczynwkdxnpdboyi.supabase.co";
    const SUPABASE_KEY = "sb_publishable_5Rzkyp1zx_b6s3EpRhf5-A_6wjsiwTS";

    // Inicializamos Supabase
    const _supabase = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

    async function cargarReplays() {
        const { data, error } = await _supabase.storage
            .from('replays') // Tu bucket debe llamarse 'replays'
            .list('', {
                limit: 50,
                sortBy: { column: 'created_at', order: 'desc' }
            });

        const listaDiv = document.getElementById('lista-replays');

        if (error) {
            listaDiv.innerHTML = `<p style="color:red">Error al conectar: ${error.message}</p>`;
            return;
        }

        if (data.length === 0) {
            listaDiv.innerHTML = "<p>No hay repeticiones grabadas todav√≠a.</p>";
            return;
        }

        listaDiv.innerHTML = ""; // Limpiar mensaje de carga

        data.forEach(file => {
            // Obtenemos la URL p√∫blica del archivo .hbr2
            const { data: urlData } = _supabase.storage.from('replays').getPublicUrl(file.name);
            const fileUrl = urlData.publicUrl;

            listaDiv.innerHTML += `
                <div class="card" onclick="abrir('${fileUrl}')">
                    <p style="color:var(--accent); font-weight:bold;">MODO 3V3</p>
                    <h3>${file.name.replace('.hbr2', '')}</h3>
                    <p>üìÖ ${new Date(file.created_at).toLocaleDateString()}</p>
                    <button class="btn-view">VER PARTIDO</button>
                </div>
            `;
        });
    }

    function abrir(url) {
        document.getElementById('player-overlay').style.display = 'flex';
        // Usamos el visor oficial de Haxball
        const visorUrl = `https://www.haxball.com/replay?url=${encodeURIComponent(url)}`;
        document.getElementById('video-content').innerHTML = `<iframe src="${visorUrl}"></iframe>`;
    }

    function cerrar() {
        document.getElementById('player-overlay').style.display = 'none';
        document.getElementById('video-content').innerHTML = "";
    }

    // Ejecutar al cargar la p√°gina
    cargarReplays();
